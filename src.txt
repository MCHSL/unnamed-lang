struct Test {
	inner

	fn get_inner() {
		self.inner
	}

	fn set_inner(x) {
		self.inner = x
	}

	fn map(f) {
		self.inner = f(self.inner)
	}

	fn call_inner(x) {
		self.inner(x)
	}
}

let test = {
	let a = 5

	new Test {
		inner: |x| x + a
	}
}

print(test.get_inner())
test.set_inner(10)
print(test.get_inner())
